How do you implement retry logic for a failed API callout in Apex? ADVANCED INTG▼ 
public class RetryableCalloutService {

    private static final Integer MAX_RETRIES = 3;
    private static final Integer RETRY_DELAY_MS = 2000; // 2 seconds

    public static HttpResponse callWithRetry(HttpRequest req) {
        Http http = new Http();
        HttpResponse res;
        Exception lastError;

        for(Integer attempt = 1; attempt <= MAX_RETRIES; attempt++) {
            try {
                res = http.send(req);

                // Retry on 429 (rate limit) or 503 (service unavailable)
                if(res.getStatusCode() == 429 || res.getStatusCode() == 503) {
                    System.debug('Attempt ' + attempt + ' got ' + res.getStatusCode() + '. Retrying...');
                    if(attempt < MAX_RETRIES) {
                        // Wait before retry (exponential backoff)
                        Long waitTime = RETRY_DELAY_MS * attempt;
                        Long endTime = System.currentTimeMillis() + waitTime;
                        while(System.currentTimeMillis() < endTime) {} // busy wait (limited use)
                        continue;
                    }
                }

                // Success or non-retryable error
                return res;

            } catch(CalloutException e) {
                lastError = e;
                System.debug('Attempt ' + attempt + ' callout exception: ' + e.getMessage());
                if(attempt == MAX_RETRIES) throw e;
            }
        }
        return res;
    }
}

How do you expose an Apex class as a REST API endpoint — complete code ADVANCED INTG▼ 
// URL: https://[instance]/services/apexrest/v1/loans/
@RestResource(urlMapping='/v1/loans/*')
global with sharing class LoanRestService {

    @HttpGet
    global static LoanResponse getLoan() {
        RestRequest req = RestContext.request;
        String loanId = req.requestURI.substring(req.requestURI.lastIndexOf('/')+1);

        try {
            Loan__c loan = [SELECT Id, LoanAmount__c, Status__c, BorrowerName__c
                            FROM Loan__c WHERE Id = :loanId WITH USER_MODE];
            return new LoanResponse(true, loan, null);
        } catch(QueryException e) {
            RestContext.response.statusCode = 404;
            return new LoanResponse(false, null, 'Loan not found: ' + loanId);
        }
    }

    @HttpPost
    global static LoanResponse createLoan() {
        try {
            String body = RestContext.request.requestBody.toString();
            LoanRequest loanReq = (LoanRequest) JSON.deserialize(body, LoanRequest.class);

            Loan__c newLoan = new Loan__c(
                LoanAmount__c = loanReq.loanAmount,
                BorrowerName__c = loanReq.borrowerName,
                LoanType__c = loanReq.loanType,
                Status__c = 'Application'
            );
            insert newLoan;

            RestContext.response.statusCode = 201; // Created
            return new LoanResponse(true, newLoan, null);
        } catch(Exception e) {
            RestContext.response.statusCode = 400; // Bad Request
            return new LoanResponse(false, null, e.getMessage());
        }
    }

    @HttpDelete
    global static LoanResponse deleteLoan() {
        String loanId = RestContext.request.requestURI.substringAfterLast('/');
        delete [SELECT Id FROM Loan__c WHERE Id = :loanId];
        RestContext.response.statusCode = 204; // No Content
        return null;
    }

    global class LoanRequest {
        global Decimal loanAmount;
        global String borrowerName;
        global String loanType;
    }

    global class LoanResponse {
        global Boolean success;
        global Object data;
        global String error;
        global LoanResponse(Boolean s, Object d, String e) { success=s; data=d; error=e; }
    }
}
How do you implement OAuth 2.0 JWT Bearer Flow for server-to-server integrations?ADV INTEGRATION▼

JWT Bearer Flow allows system-to-system authentication without any user interaction — the app proves its identity with a signed certificate.

// Setup: Connected App → Use Digital Signature → upload certificate
// Pre-authorize integration user: Connected App → Manage → Add Profile/User

// Generate JWT and exchange for access token
public class JWTBearerAuth {
  public static String getAccessToken() {
    // Build JWT header
    String header = EncodingUtil.base64Encode(Blob.valueOf('{"alg":"RS256","typ":"JWT"}'));

    // Build JWT claims
    Long now = DateTime.now().getTime() / 1000;
    String claims = JSON.serialize(new Map<String,Object>{
      'iss' => 'YOUR_CONNECTED_APP_CLIENT_ID',
      'sub' => '[email protected]', // pre-authorized user
      'aud' => 'https://login.salesforce.com',
      'exp' => now + 300 // expires in 5 minutes
    });
    String claimsEncoded = EncodingUtil.base64Encode(Blob.valueOf(claims));

    // Sign with private key from Certificate and Key Store
    String signingInput = header + '.' + claimsEncoded;
    Blob signature = Crypto.sign('RSA-SHA256', Blob.valueOf(signingInput),
      EncodingUtil.base64Decode(getCertificatePrivateKey()));
    String jwt = signingInput + '.' + EncodingUtil.base64Encode(signature);

    // Exchange JWT for access token
    HttpRequest req = new HttpRequest();
    req.setEndpoint('https://login.salesforce.com/services/oauth2/token');
    req.setMethod('POST');
    req.setBody('grant_type=urn:ietf:params:oauth:grant-type:jwt-bearer&assertion=' + jwt);
    HttpResponse res = new Http().send(req);
    Map<String,Object> result = (Map<String,Object>) JSON.deserializeUntyped(res.getBody());
    return (String) result.get('access_token');
  }
}

// Version in URL path — cleanest approach

}

@RestResource(urlMapping='/v2/loans/*')
global class LoanResourceV2 {
  @HttpGet global static LoanResponseV2 doGet() { /* v2 adds new fields, never removes */ }
}

// Version in Accept header — single endpoint, versioned response
@RestResource(urlMapping='/loans/*')
global class LoanResource {
  @HttpGet global static Object doGet() {
    String version = RestContext.request.headers.get('API-Version') ?? 'v1';
    if (version == 'v2') return buildV2Response();
    return buildV1Response(); // default to v1
  }
}
// Rules for backward compatibility:
// NEVER remove a field from response — add deprecation notice in docs, set to null
// NEVER change a field's type — add a new field with the new type
// ALWAYS version breaking changes — new path or Accept header version
API contract

External systems depend on your response shape. Breaking changes without versioning = production incidents for your API consumers. Treat your API like a public interface — additive only.

What is idempotency in API design — how do you implement it in Salesforce?ADV INTEGRATION▼

Idempotency means calling the same API multiple times produces the same result as calling it once. Critical for retry safety.

// Idempotency key pattern — client sends unique key per logical operation
@RestResource(urlMapping='/v1/loans')
global class LoanCreateResource {
  @HttpPost
  global static LoanCreateResponse doPost() {
    String idempKey = RestContext.request.headers.get('Idempotency-Key');
    if (String.isBlank(idempKey)) {
      RestContext.response.statusCode = 400;
      return new LoanCreateResponse(null, 'Idempotency-Key header required');
    }

    // Check if we already processed this key
    List<Idempotency_Record__c> existing = [
      SELECT Response_Body__c FROM Idempotency_Record__c WHERE Key__c=:idempKey LIMIT 1
    ];
    if (!existing.isEmpty()) {
      // Return cached response — safe to call multiple times
      return (LoanCreateResponse) JSON.deserialize(existing[0].Response_Body__c, LoanCreateResponse.class);
    }

    // Process the request
    Loan__c newLoan = createLoan(RestContext.request.requestBody.toString());
    LoanCreateResponse response = new LoanCreateResponse(newLoan.Id, 'Created');

    // Store the idempotency record with TTL (clean up after 24 hours via scheduled job)
    insert new Idempotency_Record__c(Key__c=idempKey, Response_Body__c=JSON.serialize(response),
      Expires_At__c=DateTime.now().addHours(24));
    RestContext.response.statusCode = 201;
    return response;
  }
}
What is HMAC signature validation — how do you secure inbound webhooks?ADV INTEGRATION▼

Webhooks from external systems (DocuSign, Twilio, Stripe) can be spoofed. HMAC-SHA256 signature validation proves the request genuinely came from the expected sender.

@RestResource(urlMapping='/v1/docusignWebhook')
global class DocuSignWebhookResource {
  @HttpPost
  global static void doPost() {
    String payload = RestContext.request.requestBody.toString();
    String receivedSig = RestContext.request.headers.get('X-DocuSign-Signature-1');
    String sharedSecret = 'YOUR_HMAC_SECRET'; // stored in Custom Setting or Named Credential

    // Compute expected HMAC-SHA256
    Blob hmac = Crypto.generateMac('HmacSHA256', Blob.valueOf(payload), Blob.valueOf(sharedSecret));
    String expectedSig = EncodingUtil.base64Encode(hmac);

    // Compare — timing-safe comparison (avoid timing attacks)
    if (receivedSig != expectedSig) {
      RestContext.response.statusCode = 401;
      return;
    }

    // Signature valid — process the webhook
    Map<String,Object> body = (Map<String,Object>) JSON.deserializeUntyped(payload);
    String envelopeId = (String) body.get('envelopeId');
    String status = (String) body.get('status');

    if (status == 'completed') System.enqueueJob(new DocuSignCompletedQueueable(envelopeId));
  }
}
How do you implement request/response logging for all outbound callouts?ADV INTEGRATION▼
// Callout logger — wraps every outbound request
public class CalloutLogger {
  public static HttpResponse send(HttpRequest req, String integrationName) {
    DateTime startTime = DateTime.now();
    HttpResponse res;
    String error = null;

    try {
      res = new Http().send(req);
      return res;
    } catch (CalloutException e) {
      error = e.getMessage();
      throw e; // re-throw after logging
    } finally {
      // Always log — success or failure
      Callout_Log__c log = new Callout_Log__c(
        Integration__c = integrationName,
        Endpoint__c    = req.getEndpoint(),
        Method__c      = req.getMethod(),
        Request_Body__c = req.getBody()?.abbreviate(32768), // field length limit
        Status_Code__c  = res != null ? res.getStatusCode() : null,
        Response_Body__c = res != null ? res.getBody()?.abbreviate(32768) : null,
        Error__c        = error,
        Duration_Ms__c  = DateTime.now().getTime() - startTime.getTime()
      );
      // Use Database.insert to avoid rollback cancelling the log
      Database.insert(log, false);
    }
  }
}

What is the outbox pattern — how do you guarantee exactly-once delivery?ADV INTEGRATION▼

The Outbox pattern ensures an event/message is never lost: write to an outbox table in the same transaction as your data change, then a separate process reliably delivers from the outbox.

// Step 1: Write loan update AND outbox record in same transaction (atomic)
trigger LoanTrigger on Loan__c (after update) {
  List<Integration_Outbox__c> outbox = new List<Integration_Outbox__c>();
  for (Loan__c l : Trigger.new) {
    if (l.Status__c != Trigger.oldMap.get(l.Id).Status__c) {
      outbox.add(new Integration_Outbox__c(
        Loan__c=l.Id, Event_Type__c='STATUS_CHANGE',
        Payload__c=JSON.serialize(new Map<String,Object>{'loanId'=>l.Id,'status'=>l.Status__c}),
        Status__c='Pending', Created_At__c=DateTime.now()
      ));
    }
  }
  if (!outbox.isEmpty()) insert outbox;
}

// Step 2: Scheduled job reliably delivers from outbox (runs every 5 min)
public class OutboxDeliveryJob implements Schedulable {
  public void execute(SchedulableContext ctx) {
    Database.executeBatch(new OutboxDeliveryBatch(), 10); // small scope for callout limit
  }
}
public class OutboxDeliveryBatch implements Database.Batchable<SObject>, Database.AllowsCallouts {
  public Database.QueryLocator start(Database.BatchableContext bc) {
    return Database.getQueryLocator([SELECT Id, Payload__c, Retry_Count__c FROM Integration_Outbox__c
      WHERE Status__c='Pending' AND Retry_Count__c < 3 ORDER BY Created_At__c ASC]);
  }
  public void execute(Database.BatchableContext bc, List<SObject> scope) {
    for (Integration_Outbox__c rec : (List<Integration_Outbox__c>)scope) {
      try { deliverToLOS(rec.Payload__c); rec.Status__c='Delivered'; }
      catch(Exception e) { rec.Retry_Count__c++; rec.Last_Error__c=e.getMessage(); }
    }
    update scope;
  }
  public void finish(Database.BatchableContext bc) {}
}

// HttpCalloutMock interface
@isTest
global class MultiRequestMock implements HttpCalloutMock {
  Map<String, HttpCalloutMock> mocks = new Map<String, HttpCalloutMock>();

  global MultiRequestMock(Map<String, HttpCalloutMock> mocks) { this.mocks = mocks; }

  global HttpResponse respond(HttpRequest req) {
    String endpoint = req.getEndpoint();
    for (String key : mocks.keySet()) {
      if (endpoint.contains(key)) return mocks.get(key).respond(req);
    }
    throw new CalloutException('No mock for endpoint: ' + endpoint);
  }
}

  }

What is mutual TLS (mTLS) and when is it required in financial services integrations?ADV INTEGRATION▼

Mutual TLS requires BOTH sides to authenticate with certificates — server proves identity to client AND client proves identity to server. Standard TLS only authenticates the server.

  • When required — Banking regulators (OCC, FFIEC), PCI-DSS high-security APIs, Federal agency integrations, some credit bureau APIs
  • Setup in Salesforce — Generate a certificate in Setup → Certificate and Key Management. Configure Named Credential to use that certificate as client certificate.
  • Server certificate pinning — Some APIs also require pinning the server's exact certificate. Salesforce Named Credentials don't support pinning natively — requires a middleware proxy.
  • Certificate rotation — mTLS certificates expire. Build rotation reminders into your monitoring — typically 1 year. Expiry = instant outage for all callouts using that cert.
  • Testing — Use Salesforce Sandbox + named credential pointing to vendor's test environment with test certificates. Never test mTLS in production.
How do you design a reliable file transfer integration — large files to/from Salesforce?ADV INTEGRATION▼

Direct callouts have a 12MB response/request limit. For large files (loan documents, credit reports, closing packages), you need a multi-step approach.

// Pattern: Salesforce → S3 (or Azure Blob) for large files
// Step 1: Get pre-signed URL from AWS (short-lived, secure)
public class S3UploadService {
  public static String getPresignedUploadUrl(String fileName, String contentType) {
    HttpRequest req = new HttpRequest();
    req.setEndpoint('callout:AWS_API_GW/presigned-url');
    req.setMethod('POST');
    req.setBody(JSON.serialize(new Map<String,Object>{'fileName'=>fileName,'contentType'=>contentType}));
    HttpResponse res = new Http().send(req);
    Map<String,Object> result = (Map<String,Object>) JSON.deserializeUntyped(res.getBody());
    return (String) result.get('uploadUrl');
  }

  // Step 2: Upload file directly to S3 using the pre-signed URL
  public static void uploadFile(String presignedUrl, Blob fileContent, String contentType) {
    HttpRequest req = new HttpRequest();
    req.setEndpoint(presignedUrl);
    req.setMethod('PUT');
    req.setHeader('Content-Type', contentType);
    req.setBodyAsBlob(fileContent); // up to 12MB per callout
    req.setTimeout(120000); // 2 minutes for large files
    HttpResponse res = new Http().send(req);
    if (res.getStatusCode() != 200) throw new IntegrationException('S3 upload failed: '+res.getStatusCode());
  }
}
